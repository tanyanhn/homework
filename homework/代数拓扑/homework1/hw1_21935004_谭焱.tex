\begin{exc}
设$ \alpha : I \rightarrow X $ 一条道路$ \alpha(0) = x_0 $,$ e_{x_0} : I \rightarrow X $　表示$ x_0 $ 处常道路作连接

证明：　$ e_{x_0} * \alpha \simeq_p \alpha $.
\end{exc}

\begin{solution} 
	\begin{equation}
		 e_{x_0} * \alpha = 
		 \begin{cases}
		 x_0     \qquad & 0 \leq s \leq \frac{1}{2}\\
		 \alpha(2s-1)   & \frac{1}{2} < s \leq 1.
		 \end{cases}
	\end{equation}
	Define \begin{equation}
	H(s,t) = \begin{cases}
	F(s,t) = x_0   \qquad & (s,t) \in A \\
	G(s,t) = \alpha(\frac{2s +t -1}{1+t})    & (s,t) \in B.
	\end{cases}
	\end{equation}
\begin{tikzpicture}[x = 2cm, y = 2cm]
	\draw[->](-0.5,0)--(1.5,0) node[right]{s轴};
	\draw[->](0,-0.5)--(0,1.5) node[above]{t轴};
	\draw  (0,0) -- (1,0);
	\draw  (0,0) -- (0,1);
	\draw  (1,1) -- (0,1);
	\draw  (1,1) -- (1,0);
	\foreach \x in {0.5, 1}
	\draw(\x,0)node[below]{$\x$}--(\x ,0)--++(0,3pt);
	\foreach \x in {0.5, 1}
	\draw(0,\x)node[left]{$\x$}--(0,\x )--++(3pt,0);
	\draw(0.2,0.25)node{A};
	\draw(0.6,0.6)node{B};
	\draw(0,0)node[below left]{0,0};
	\draw[dashed]  (0,1) -- (0.5,0);
\end{tikzpicture}

在上图中
\begin{equation}\nonumber
	\begin{aligned}
	H(s,1) = \alpha(s) \qquad  &H(s,0) = e_{x_0} * \alpha(s) \\
	H(0,t) = F(0,t) = x_0 \qquad &H(1,t) = G(1,t) = \alpha(1) = x_1.
	\end{aligned}
\end{equation}

因为在$ A,B $ 的公共边界$ 2s + t -1 = 0 $ 上　$ F(s,t) = x_0 = \alpha(0) = G(s,t)（$
由粘合引理$ H(s,t) $ 连续，所以$ e_{x_0} * \alpha \simeq_p \alpha $.



\end{solution}

\begin{exc}
	设$ F: I \times I \rightarrow X $ 连续，令$ F(0,t) = \gamma(t), F(s,1) = \beta(s), F(1,t) = \delta(t), F(s,0) = \alpha(s), $
	
	证明：　$ \alpha \simeq_p (\gamma * \beta) * \delta^{-1} $.
\end{exc} 

\begin{solution}
	$ F(s,t) $ 在　$ I \times I $ 上的图像如下
	
	\begin{tikzpicture}[x = 2cm, y = 2cm]
	\draw[->](-0.5,0)--(1.5,0) node[right]{x轴};
	\draw[->](0,-0.5)--(0,1.5) node[above]{y轴};
	\draw  (0,0) -- (1,0);
	\draw  (0,0) -- (0,1);
	\draw  (1,1) -- (0,1);
	\draw  (1,1) -- (1,0);
	\foreach \x in {0.5, 1}
	\draw(\x,0)node[below]{$\x$}--(\x ,0)--++(0,3pt);
	\foreach \x in {0.5, 1}
	\draw(0,\x)node[left]{$\x$}--(0,\x )--++(3pt,0);
	\draw(0.8,0.2)node{A};
	\draw(0.2,0.8)node{B};
	\draw(0.5,0.5)node{F(x,y)};
	\draw(0,0)node[below left]{(0,0)};
	\draw(1,1)node[above right]{(1,1)};
	\draw[dashed]  (0,0) -- (1,1);
	\end{tikzpicture}
	\begin{tikzpicture}[x = 2cm, y = 2cm]
	\draw[->](-0.5,0)--(1.5,0) node[right]{s轴};
	\draw[->](0,-0.5)--(0,1.5) node[above]{t轴};
	\draw  (0,0) -- (1,0);
	\draw  (0,0) -- (0,1);
	\draw  (1,1) -- (0,1);
	\draw  (1,1) -- (1,0);
	\foreach \x in {0.5, 1}
	\draw(\x,0)node[below]{$\x$}--(\x ,0)--++(0,3pt);
	\foreach \x in {0.5, 1}
	\draw(0,\x)node[left]{$\x$}--(0,\x )--++(3pt,0);
	\draw(0.5,0.2)node{A};
	\draw(0.5,0.8)node{B};
	\draw(0.5,0.5)node{H(s,t)};
	\draw(0,0)node[below left]{(0,0)};
	\draw(1,1)node[above right]{(1,1)};
	\draw[dashed]  (0,0.5) -- (1,0.5);
	\end{tikzpicture}
	
	Define \begin{equation}
	H(s,t) = \begin{cases}
	S(s,t) = F(2s - 2st, 2st)   \qquad & (s,t) \in A \\
	G(s,t) = F(1-2t+2st,2s-1+2t-2st)    & (s,t) \in B.
	\end{cases}
	\end{equation}
	
	容易验证　
	\begin{equation}\nonumber
	\begin{aligned}
	H(s,1) = \gamma * \beta \qquad  H(s,0) = \alpha * \delta \\
	H(0,t) = F(0,0) = \alpha(0) = \gamma(0) \\
	H(1,t) = F(1,1) = \delta(1) = \beta(1).
	\end{aligned}
	\end{equation}
	
	容易看到图２中的虚线映射到图一的虚线上，因此由$ F $的连续性得$ H $连续，即$ \alpha * \delta \simeq_p \gamma * \beta $.又因为$ * $构成群，即$ \alpha * \delta \simeq_p \gamma * \beta \Longleftrightarrow \alpha \simeq_p \gamma * \beta * \delta^{-1} $.　
\end{solution}
